4.)SELECT e.last_name, j.job_title, d.DEPARTMENT_NAME,  (2019 - EXTRACT (YEAR FROM hire_date))year_antiguedad 
FROM employees e left join departments d on (e.department_id= d.department_id) left join jobs j on (e.job_id=j.job_id)
WHERE (2019 - EXTRACT (YEAR FROM hire_date)) IN (6,10,11,15,16,17);
